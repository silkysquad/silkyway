## `GET /api/transfers/:pda`

Returns full details of a specific Handshake escrow transfer.

### Path Parameters

<ParamField path="pda" type="string" required>
  The transfer PDA (on-chain address).
</ParamField>

### Response

<ResponseField name="transfer" type="object">
  <Expandable title="Fields">
    <ResponseField name="pda" type="string">
      Transfer PDA.
    </ResponseField>
    <ResponseField name="sender" type="string">
      Sender's wallet address.
    </ResponseField>
    <ResponseField name="recipient" type="string">
      Recipient's wallet address.
    </ResponseField>
    <ResponseField name="amount" type="number">
      Transfer amount in token units.
    </ResponseField>
    <ResponseField name="memo" type="string">
      Transfer memo.
    </ResponseField>
    <ResponseField name="status" type="string">
      Transfer status: `Active`, `Claimed`, `Cancelled`, `Rejected`, `Declined`, `Expired`.
    </ResponseField>
    <ResponseField name="pool" type="string">
      Escrow pool PDA.
    </ResponseField>
    <ResponseField name="claimableAfter" type="string">
      Earliest claim time (ISO 8601), if set.
    </ResponseField>
    <ResponseField name="claimableUntil" type="string">
      Claim deadline (ISO 8601), if set.
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      Creation timestamp.
    </ResponseField>
  </Expandable>
</ResponseField>

curl https://silk.silkyway.ai/api/transfers/4bHUkQ9bR3mFVDg2Thh3AG6sFRPqNrDJ4
